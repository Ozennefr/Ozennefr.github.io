function Base.:*(pere::Individual, mere::Individual)
    #1 points crossover
    point = rand(1:(length(pere.genotype)))
    chr1 = [pere.genotype[1:point]..., mere.genotype[1+point:end]...]
    chr2 = [mere.genotype[1:point]..., pere.genotype[1+point:end]...]
    offspring1 = Individual(0, 0, chr1, [], Inf)
    offspring2 = Individual(0, 0, chr2, [], Inf)
    return(offspring1, offspring2)
end

Building blocks

function initPopulation(popSize::Int64, chromosomeSize::Int64)
    population = Array{Individual}(popSize);
    for iInd in 1:popSize
        population[iInd] = Individual(iInd,1,rand(Bool, (chromosomeSize)),[],Inf);
    end
    return(population)
end;

function expressGenotype!(individual::Individual, problem::Problem)
    genePerVar = round(Int64,(size(individual.genotype)[1]/problem.numberOfVar))
    individual.phenotype = Array{Float64}(problem.numberOfVar)

    for iVar in 1:problem.numberOfVar
        individual.phenotype[iVar] = 0;
        for iGene in 1:genePerVar
            individual.phenotype[iVar] += individual.genotype[iGene+(iVar-1)*genePerVar]*2.0^(iGene-1)
        end
        individual.phenotype[iVar] = -problem.range[iVar] + 2.0*problem.range[iVar]/(1.-2.0.^(-genePerVar)) .*
                                    2.0^(-genePerVar).*individual.phenotype[iVar]
    end
end

function evaluateIndividual!(individual::Individual, problem::Problem)
    individual.fitness = problem.fitness(individual.phenotype...);
end

function selectIndividual(tournamentSize::Int64, ptour::Float64, pop::Array{Individual,1})
    selected = Individual(0, 0, [], [], Inf)
    fitnesses = Array{Float64}(tournamentSize)
    subPop = sample(pop, tournamentSize, replace=true)
    for j in 1:tournamentSize
        fitnesses[j] = subPop[j].fitness
    end
    notSelected = true
    while notSelected
        mxval, mxindx = findmin(fitnesses)
        if rand(Float64) < ptour
            notSelected = false
            selected = subPop[mxindx]
        elseif length(fitnesses)==1
            selected = subPop[mxindx]
        else
            deleteat!(fitnesses, mxindx)
        end
    end
    return(selected)
end

function selectIndividuals(nInd::Int64, tournamentSize::Int64, ptour::Float64, pop::Array{Individual,1})
    selected = Array{Individual,1}(nInd)
    for i in 1:nInd
        selected[i] = selectIndividual(tournamentSize, ptour, pop)
    end
    return(selected)
end

function mutate!(ind::Individual, pmut::Float64)
    for i in 1:length(ind.genotype)
        if rand(Float64) < pmut
            ind.genotype[i] = (ind.genotype[i]+1)%2 #bitflip
        end
    end
end;

Plotting progress

function plotPopulation(pop::Array{Individual,1}, problem::Problem, fitnessHistory)

    nVar = problem.numberOfVar
    nIndiv = length(pop)

    ph = Array{Float64,2}(nIndiv, nVar)
    f = Array{Float64,1}(nIndiv)
    g = Array{Int64,1}(nIndiv)
    for i in 1:nIndiv
        ph[i,:] = pop[i].phenotype
        f[i] = pop[i].fitness
        g[i] = pop[i].generation
    end

    y = [histogram(ph[:,i], title="y$i", nbins = 20) for i in 1:nVar]
    y_plot = plot(y..., layout=(1,nVar))

    fp = histogram(f, title="fitness distribution", nbins = 20)
    gp = histogram(g, title="generation distribution", nbins = 20)
    statp = plot(fp, gp, layout = (2,1))
    fh = plot(fitnessHistory, title="fitness history")
    fplot = plot(fh, statp, layout=(1,2))
    plot(y_plot, fplot, layout=(2,1))
    #scatter(ph[:,1],ph[:,2],zcolor=f)
    IJulia.clear_output(true)
    plot!(legend=false) |> display
end

The next generation

function makeNewGeneration(pop::Array{Individual,1}, gen::Int64, ptour::Float64, 
                            tournamentSize::Int64, carryOver::Float64, pmut::Float64,
                            elitism::Bool, bestInd::Int64, pcross::Float64)
    popSize = length(pop)
    newGen = Array{Individual}(popSize)
    if carryOver != 0.
        newGen[1:round(Int, carryOver*popSize)] = selectIndividuals(round(Int, carryOver*popSize), tournamentSize, ptour, pop)
    end
    for i in round(Int, carryOver*popSize)+1:2:popSize
        if rand(Float64) < pcross
            offspring1, offspring2  = *(selectIndividuals(2, tournamentSize, ptour, pop)...)
        else
            parents = selectIndividuals(2, tournamentSize, ptour, pop)
            offspring1, offspring2 = (parents[1], parents[2])
        end
        offspring1.id = i+gen*popSize
        offspring1.generation = 1+gen
        offspring2.id = i+1+gen*popSize
        offspring2.generation = 1+gen
        mutate!(offspring1, pmut)
        mutate!(offspring2, pmut)
        newGen[i] = offspring1
        newGen[i+1] = offspring2
    end
    if elitism
        newGen[1]=pop[bestInd]
    end
    return(newGen)
end
;

Genetic algorithm

function geneticOptimisation(problem::Problem; chromosomeSize::Int64=128, 
                            popSize::Int64=10000, years::Int64=100, 
                            ptour::Float64=0.8, tournamentSize::Int64=5,
                            carryOver::Float64 = 0., pmut::Float64 = 0.1,
                            elitism::Bool = false, pcross::Float64 = 0.8)
    pop = initPopulation(popSize, chromosomeSize)
    fitnessHistory = Array{Float64,1}(years)
    bestFitness = Inf
    bestIndiv = 0
    fill!(fitnessHistory, 0)
    for gen in 1:years
        bestFitness = Inf
        bestIndiv = 0
        for i in 1:popSize
            expressGenotype!(pop[i], problem);
            evaluateIndividual!(pop[i], problem);
            if pop[i].fitness < bestFitness
                bestFitness = pop[i].fitness
                bestIndiv = i
            end
        end
        fitnessHistory[gen] = bestFitness
        if gen%100 == 1
            plotPopulation(pop, problem, fitnessHistory)
        end
        if gen <years
            pop = makeNewGeneration(pop, gen, ptour, tournamentSize, carryOver, pmut, elitism, bestIndiv, pcross)
        end
    end
    return(pop, bestFitness, bestIndiv)
end
;
